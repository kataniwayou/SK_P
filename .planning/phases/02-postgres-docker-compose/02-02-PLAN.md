**Step 3 — Poll for `healthy` status**

The healthcheck cadence is 5s interval × up to 10 retries + 5s start_period = ~55s upper bound. Poll for 90s to give comfortable margin on a cold dev box.

```powershell
$ready = $false
$pollStart = Get-Date
$psOutput = $null
for ($i = 0; $i -lt 18; $i++) {  # 18 × 5s = 90s
    Start-Sleep -Seconds 5
    $psOutput = docker compose ps postgres --format json 2>&1
    if ($LASTEXITCODE -ne 0) { continue }

    # `docker compose ps --format json` emits one JSON object per line in v2.20+; older versions emit an array.
    # Parse defensively: look for either Health=healthy field or the legacy Status string containing '(healthy)'.
    $hasHealthy = $false
    foreach ($line in ($psOutput -split "`n")) {
        $trimmed = $line.Trim()
        if (-not $trimmed) { continue }
        try {
            $obj = $trimmed | ConvertFrom-Json
            if ($obj -is [System.Array]) {
                foreach ($e in $obj) { if ($e.Health -eq 'healthy' -or $e.Status -match '\(healthy\)') { $hasHealthy = $true; break } }
            } else {
                if ($obj.Health -eq 'healthy' -or $obj.Status -match '\(healthy\)') { $hasHealthy = $true }
            }
        } catch {}
        if ($hasHealthy) { break }
    }

    if ($hasHealthy) {
        $ready = $true
        $pollDuration = (Get-Date) - $pollStart
        Write-Host "PASS: postgres reported healthy after $($pollDuration.TotalSeconds)s of polling"
        break
    }
}

if (-not $ready) {
    Write-Host '--- docker compose ps postgres (final state) ---'
    docker compose ps postgres 2>&1
    Write-Host '--- docker compose logs postgres (last 60 lines) ---'
    docker compose logs --tail=60 postgres 2>&1
    Write-Error 'FAIL: postgres did not report healthy within 90s. Diagnose using the logs above. Common causes: pg_isready env-var interpolation broken (Pitfall 24 — verify `$$` escaping), port collision on 5433 (Pitfall 25 — `Get-NetTCPConnection -LocalPort 5433`), Docker Desktop low resources.'
    exit 2
}
```
