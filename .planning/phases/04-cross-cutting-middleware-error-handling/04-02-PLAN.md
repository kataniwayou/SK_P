    ```csharp
    using System.Net.Http.Json;
    using System.Text.Json;
    using Microsoft.EntityFrameworkCore;
    using Xunit;

    namespace BaseApi.Tests.Middleware;

    public sealed class ConcurrencyTokenTests : IClassFixture<PostgresFixture>
    {
        private readonly PostgresFixture _fixture;

        public ConcurrencyTokenTests(PostgresFixture fixture) => _fixture = fixture;

        [Fact]
        public async Task Test_RacingWrites_Produce_409_WithGenericMessage_NoXminLeak()
        {
            var ct = TestContext.Current.CancellationToken;
            using var factory = new WebAppFactory(_fixture.ConnectionString);
            using var client = factory.CreateClient();

            var options = new DbContextOptionsBuilder<TestErrorDbContext>()
                .UseNpgsql(_fixture.ConnectionString)
                .UseSnakeCaseNamingConvention()
                .Options;

            // Seed parent row.
            var parentId = Guid.NewGuid();
            await using (var db = new TestErrorDbContext(options))
            {
                await db.Database.EnsureCreatedAsync(ct);
                db.Parents.Add(new TestParentEntity { Id = parentId, Name = "p-initial", Version = "1.0.0" });
                await db.SaveChangesAsync(ct);
            }

            // First writer (out-of-band) — advances xmin.
            await using (var db = new TestErrorDbContext(options))
            {
                var p = await db.Parents.FirstAsync(p => p.Id == parentId, ct);
                p.Name = "p-firstWriter";
                await db.SaveChangesAsync(ct);
            }

            // Second writer via HTTP endpoint — the controller loaded the row BEFORE the first
            // writer's commit (in this single-process test, we simulate by having the endpoint
            // load + mutate AFTER first writer; since endpoint uses its own DbContext, it sees
            // the row with the OLD xmin from its tracking snapshot. NOTE: in an in-process test
            // we have to fake the staleness — simplest path: the endpoint loads with one xmin,
            // we directly UPDATE the row from a separate connection to advance xmin, then the
            // endpoint's SaveChangesAsync fires DbUpdateConcurrencyException). The TestController
            // /concurrency endpoint above does load-mutate-save; we need to advance xmin BETWEEN
            // the endpoint's load and its save. The cleanest pattern: invoke the endpoint, but
            // first advance xmin (already done above as "first writer"), so when the endpoint
            // loads it gets the CURRENT xmin and its update succeeds. To force a concurrency
            // exception we need TWO endpoint invocations OR to load the entity from a DbContext
            // in the test BEFORE calling the endpoint.
            //
            // Simpler: do the racing in-test via two DbContexts (skip HTTP for the race itself),
            // then make a separate HTTP call to /test/concurrency to verify the HTTP path
            // wraps the exception correctly. That isn't a true HTTP race, but it tests the
            // mapping which is what the SC needs.

            await using var stale = new TestErrorDbContext(options);
            var p1 = await stale.Parents.FirstAsync(p => p.Id == parentId, ct);

            await using (var freshWriter = new TestErrorDbContext(options))
            {
                var p2 = await freshWriter.Parents.FirstAsync(p => p.Id == parentId, ct);
                p2.Name = "p-secondWriter";
                await freshWriter.SaveChangesAsync(ct);
            }

            // Now stale context's xmin is behind — modifying + saving raises DbUpdateConcurrencyException.
            p1.Name = "p-thirdWriter-stale";
            var ex = await Assert.ThrowsAsync<DbUpdateConcurrencyException>(
                async () => await stale.SaveChangesAsync(ct));
            Assert.NotNull(ex);

            // Now verify the HTTP-level mapping: hit the concurrency endpoint (which does the same
            // load+save pattern). Because the row was just updated, the endpoint's SaveChangesAsync
            // either succeeds (if no other concurrent writer) OR fails with DbUpdateConcurrencyException
            // (if we advance xmin again between its load and save). The HTTP test below just
            // confirms the response shape when DbUpdateConcurrencyException IS thrown — which we
            // can force by having a second DbContext mutate after the endpoint's load. For
            // simplicity, the WebAppFactory's TestController uses load+save in one method, so
            // we trigger the race by hitting the endpoint TWICE concurrently:

            using var task1 = Task.Run(() => client.PostAsync($"/test/concurrency?id={parentId}", null, ct));
            using var task2 = Task.Run(() => client.PostAsync($"/test/concurrency?id={parentId}", null, ct));
            var responses = await Task.WhenAll(task1, task2);

            // At least one should be 200, at most one 409.
            var conflict = responses.FirstOrDefault(r => r.StatusCode == HttpStatusCode.Conflict);
            if (conflict is not null)
            {
                var body = await conflict.Content.ReadAsStringAsync(ct);
                using var doc = JsonDocument.Parse(body);
                Assert.Equal(409, doc.RootElement.GetProperty("status").GetInt32());
                Assert.Equal("The resource was modified by another request; reload and retry.",
                    doc.RootElement.GetProperty("detail").GetString());

                // T-04-XMIN: ensure no numeric xmin substring leaks.
                Assert.DoesNotMatch(@"""xmin""\s*:\s*\d+", body);
                Assert.DoesNotContain("xmin", body);  // generic msg has no xmin token
            }
            else
            {
                // Race didn't fire — degenerate to direct unit-style test of the EF behavior path.
                // The earlier Assert.ThrowsAsync<DbUpdateConcurrencyException> proves the EF layer
                // raises it; the HTTP-mapping branch is covered by FallbackExceptionHandler since
                // we did NOT see a 409 — which would be a regression. So if no 409 fires across
                // two concurrent POSTs, fail the test.
                Assert.Fail("Concurrency race did not produce a 409 — DbUpdateExceptionHandler concurrency branch may be broken.");
            }
        }
    }
    ```
