```csharp
using MassTransit;
using Messaging.Contracts;
using Messaging.Contracts.Projections;
using Microsoft.Extensions.Options;
using StackExchange.Redis;

namespace Keeper.Recovery;

public enum ProbeOutcome { Recovered, GaveUp }

/// <summary>
/// PROBE-01/02 (D-02/03/04): bounded L2 read+write probe loop, awaited inside Consume. Each iteration does a
/// READ (skp:data:{entryId} — value need NOT exist) AND a WRITE-then-delete of the scratch key
/// (skp:keeper:probe:{h}, short TTL = crash net-zero net). Success ONLY if BOTH ops complete with no Redis
/// exception. A RedisException (the superset of RedisConnectionException + RedisTimeoutException) = keep
/// looping after a DelaySeconds delay. Do NOT catch Exception (a genuine bug must not masquerade as "down").
/// The IConnectionMultiplexer is the SAME singleton AddBaseConsole already registers (mirrors ResultConsumer).
/// </summary>
public sealed class L2ProbeRecovery(IConnectionMultiplexer redis, IOptions<ProbeOptions> opts)
{
    public async Task<ProbeOutcome> RunAsync(IExecutionCorrelated inner, CancellationToken ct)
    {
        var db = redis.GetDatabase();
        var max = opts.Value.MaxAttempts;
        for (var attempt = 0; attempt < max; attempt++)
        {
            try
            {
                _ = await db.StringGetAsync(L2ProjectionKeys.ExecutionData(inner.EntryId));   // READ — value need NOT exist (D-02)
                var scratch = (RedisKey)L2ProjectionKeys.KeeperProbe(inner.H);
                await db.StringSetAsync(scratch, "1", expiry: TimeSpan.FromSeconds(30));        // WRITE w/ short TTL (D-03)
                await db.KeyDeleteAsync(scratch);                                               // then delete (net-zero)
                return ProbeOutcome.Recovered;                                                  // both ops, no exception (D-02)
            }
            catch (RedisException)   // RedisConnectionException + RedisTimeoutException both derive from this
            {
                if (attempt + 1 < max)
                    await Task.Delay(TimeSpan.FromSeconds(opts.Value.DelaySeconds), ct);
            }
        }
        return ProbeOutcome.GaveUp;
    }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Redis (L2) ↔ Keeper probe | half-open Redis (read OK, write fails) must count as "down" — the probe write closes this gap |
| Keeper → origin endpoint (re-inject) | Keeper Sends a verbatim inner message back onto the bus |
| Keeper → keeper-dlq (park) | Keeper Sends the original Fault<T> envelope (carries Exceptions[] payload) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-36-04 | Spoofing/Tampering | re-injected inner message | accept | Re-inject is verbatim (same H); the receiver's surviving Phase-31 `flag[H]` gate collapses duplicates (PROBE-06). No new auth surface (internal bus). |
| T-36-05 | DoS | half-open Redis swallowed as "up" | mitigate | The probe requires BOTH a read AND a write-then-delete to succeed (D-02); a write-failing half-open Redis keeps the loop running, so recovery is not declared prematurely. Asserted by `Probe_RequiresReadAndWrite`. |
| T-36-06 | DoS | `catch (Exception)` masking a code bug as "down" | mitigate | The loop catches `RedisException` ONLY (RESEARCH Anti-Pattern); a genuine bug propagates → Immediate(N) → DLQ-1, never loops 60s. Asserted by the `catch (Exception` == 0 grep. |
| T-36-07 | Information disclosure | parked Fault<T> envelope carries Exceptions[] / payload | accept | Operators draining keeper-dlq already have broker access; same exposure as the existing `_error`. Information log surfaces only ExceptionType+Message (no stack frames — T-35-05, kept). |
| T-36-08 | DoS | poison-message re-inject loop (genuinely bad message re-faults forever) | accept | Pre-existing milestone model limitation (Keeper recovers transient L2 faults only); a non-L2 fault re-faults → DLQ-1 TTL forensics. Not a Phase-36 regression (RESEARCH Security Domain). |
</threat_model>
