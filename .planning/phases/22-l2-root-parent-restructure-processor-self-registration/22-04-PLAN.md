    ```csharp
    using System.Text.Json;
    using BaseApi.Service.Features.Orchestration.Projection;
    using Messaging.Contracts.Projections;
    using StackExchange.Redis;

    namespace BaseApi.Service.Features.Orchestration.Validation;

    /// <summary>
    /// Processor existence + timestamp-liveness gate (PROC-LIVE-01). ASYNC — reads each participating
    /// processor's self-registered L2 entry (skp:{procId}), unlike the three sync gates. Absent or stale
    /// (timestamp + interval*2 &lt;= now) throws OrchestrationValidationException.ProcessorNotLive → 422,
    /// the same contract as SchemaEdgeValidator. `interval` is interpreted in SECONDS and sourced from the
    /// entry (NOT the hardcoded 0). Slots AFTER the sync trio and BEFORE UpsertAsync (D-15). The only Redis
    /// reads are the per-processor liveness GETs (external self-registered data, not in the L1 snapshot).
    /// </summary>
    internal sealed class ProcessorLivenessValidator
    {
        private readonly IConnectionMultiplexer _multiplexer;
        private readonly TimeProvider _clock;

        public ProcessorLivenessValidator(IConnectionMultiplexer multiplexer, TimeProvider clock)
        {
            _multiplexer = multiplexer ?? throw new ArgumentNullException(nameof(multiplexer));
            _clock       = clock ?? throw new ArgumentNullException(nameof(clock));
        }

        public async Task ValidateAsync(WorkflowGraphSnapshot snapshot, CancellationToken ct)
        {
            var db = _multiplexer.GetDatabase();
            var now = _clock.GetUtcNow().UtcDateTime;   // mirrors RedisProjectionWriter.cs:60
            foreach (var proc in snapshot.Processors.Values)
            {
                var raw = await db.StringGetAsync(RedisProjectionKeys.Processor(proc.Id));
                if (raw.IsNullOrEmpty)
                    throw OrchestrationValidationException.ProcessorNotLive(proc.Id, "absent");

                var projection = JsonSerializer.Deserialize<ProcessorProjection>(raw!)!;
                var liveness = projection.Liveness;
                // D-16 — interval in SECONDS, from the entry (not hardcoded 0): timestamp + interval*2 > now.
                var deadline = liveness.Timestamp.AddSeconds(liveness.Interval * 2);
                if (deadline <= now)
                    throw OrchestrationValidationException.ProcessorNotLive(proc.Id, "stale");
            }
        }
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| external processor → L2 → validator | Self-registered `ProcessorProjection` JSON is untrusted input GET-then-deserialized at Start |
| validator → HTTP response | Validation failure must map to the SAME 422 contract — never 500 or bypass |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-22-09 | Spoofing | forged/oversized self-registered liveness JSON | mitigate | The validator deserializes into the strongly-typed `ProcessorProjection` record (camelCase-pinned); a malformed timestamp/interval cannot bypass the `timestamp + interval*2 > now` check — a far-past or zero value yields `deadline <= now` → "stale" → 422 (fail-safe to reject, never fail-open to allow). |
| T-22-10 | Tampering | absent / stale processor entry | mitigate | Absence throws `ProcessorNotLive(.., "absent")` → 422; staleness throws `.. "stale"` → 422. No code path lets an unhealthy/absent processor pass the gate. |
| T-22-11 | Denial of service (fail-open) | deserialize throwing on malformed JSON → 500 instead of 422 | accept | Malformed (non-deserializable) JSON throws `JsonException` → FallbackExceptionHandler → 500 with no body leak (Phase 4 customizer). This is a 500, not a bypass — the Start is still rejected (fail-safe). A dedicated 422 for malformed-vs-stale is out of scope this phase (self-registration write path is external/Deferred); accepted as low risk since the only writers are trusted external processors. |
| T-22-12 | Information disclosure | 422 offending payload | mitigate | `ProcessorLivenessOffending` carries only the processor `Guid` + a fixed enum-like reason string ("absent"/"stale") — no stack traces, connection strings, or internal type names (consistent with the T-14-02 guard documented on the exception class). |
| T-22-13 | Information disclosure | Redis fault op-tag | mitigate | A `RedisException` during the GETs is tagged `redisOp = "UpsertAsync"` (OBSERV-REDIS-03) and surfaces only the stable op name in the 500 body — never connection details. |
</threat_model>
