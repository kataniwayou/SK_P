/// <summary>
/// KEEP-02 binding-shape proof (inverse of FanOutBroadcastTests). ONE shared competing-consumer
/// endpoint + ONE consumer type â†’ a single publish is consumed exactly ONCE (load-balance), not
/// broadcast. The in-memory harness has a single endpoint instance, so this proves the binding
/// SHAPE (shared, not fan-out); true cross-replica distribution is the live-stack operator smoke (Task 3).
/// </summary>
public sealed class KeeperRoundRobinTests
{
    [Fact]
    public async Task One_Publish_Is_Delivered_To_Exactly_One_Consumer_On_Shared_Endpoint()
    {
        var ct = TestContext.Current.CancellationToken;
        await using var provider = new ServiceCollection()
            .AddLogging()
            .AddMassTransitTestHarness(x =>
            {
                // Plain AddConsumer + the stable EndpointName from PlaceholderConsumerDefinition =>
                // ONE shared queue (NO InstanceId/Temporary). RetryOptions must be registered so the
                // definition's IOptions<RetryOptions> ctor resolves.
                x.AddConsumer<PlaceholderConsumer, PlaceholderConsumerDefinition>();
                x.UsingInMemory((c, cfg) => cfg.ConfigureEndpoints(c));
            })
            .AddOptions<RetryOptions>().Configure(o => o.Limit = 3).Services
            .BuildServiceProvider(true);

        var harness = provider.GetRequiredService<ITestHarness>();
        await harness.Start();
        try
        {
            await harness.Bus.Publish(new KeeperPlaceholder { CorrelationId = NewId.NextGuid() }, ct);
            var consumer = harness.GetConsumerHarness<PlaceholderConsumer>();
            Assert.True(await consumer.Consumed.Any<KeeperPlaceholder>(ct));
            Assert.Equal(1, consumer.Consumed.Select<KeeperPlaceholder>(ct).Count());   // LOAD-BALANCE: exactly one
        }
        finally { await harness.Stop(ct); }
    }
}
