```powershell
# Run Release UNCONDITIONALLY
$null = & dotnet build --configuration Release --no-restore --verbosity normal 2>&1 | Tee-Object -Variable releaseOut
$releaseExit = $LASTEXITCODE
$releaseFailed = ($releaseExit -ne 0) -or ($releaseOut -join "`n" -match 'Warning')

# Run Debug UNCONDITIONALLY (independent of Release outcome)
$null = & dotnet build --configuration Debug --no-restore --verbosity normal 2>&1 | Tee-Object -Variable debugOut
$debugExit = $LASTEXITCODE
$debugFailed = ($debugExit -ne 0) -or ($debugOut -join "`n" -match 'Warning')

# Aggregate: fail if EITHER failed
if ($releaseFailed -or $debugFailed) {
    Write-Host "BUILD AGGREGATE FAIL: ReleaseExit=$releaseExit ReleaseFailed=$releaseFailed DebugExit=$debugExit DebugFailed=$debugFailed"
    Write-Host "--- Release output (last 50 lines) ---"
    $releaseOut | Select-Object -Last 50
    Write-Host "--- Debug output (last 50 lines) ---"
    $debugOut | Select-Object -Last 50
    exit 1
}
Write-Host "BUILD AGGREGATE PASS: Release exit=0 zero-warnings AND Debug exit=0 zero-warnings"
```

```powershell
$proc = Start-Process -FilePath "dotnet" `

    -RedirectStandardOutput $logFile `
    -RedirectStandardError $errFile `
    -NoNewWindow -PassThru

# Poll for readiness — up to 30 seconds. A 404 from GET / proves the host is up (Phase 1 has no routes).
$ready = $false

    Start-Sleep -Seconds 1

    # Bail out early if the process exited before binding (config crash, port conflict, etc.)
    if ($proc.HasExited) {
        Write-Host "FAIL: dotnet process exited before host ready (exit code $($proc.ExitCode))"
        Write-Host "--- stdout ---"
        Get-Content $logFile -ErrorAction SilentlyContinue
        Write-Host "--- stderr ---"
        Get-Content $errFile -ErrorAction SilentlyContinue
        exit 1

        Write-Host "FAIL: GET / returned 2xx — Phase 1 must have no routes registered"
        Stop-Process -Id $proc.Id -Force -ErrorAction SilentlyContinue
        exit 2

        if ($status -eq 404) {
            $ready = $true

    } catch {
```
